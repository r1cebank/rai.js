    class DataSource
  constructor: (db_type, db_url) ->
    @datasource = require('../config/datasource.json').datasource[db_type]
    @type = db_type
    @url = db_url
    @ds_connection = require(@datasource.connection)()
    @ds_query = require(@datasource.query)()
    @connection = @ds_connection.connect @url
